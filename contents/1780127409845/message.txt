end

local function function22(function25, function24)
    if function25 == function24 then return true end
    local UNIQUE_TYPES = {
        ["function"] = true,
        ["table"] = true,
        ["userdata"] = true,
        ["thread"] = true
    }
    for index9, index8 in pairs(function25) do
        if UNIQUE_TYPES[type(index8)] then
            if type(function24[index9]) ~= type(index8) then return false end
        elseif function24[index9] ~= index8 then
            return false
        end 
    end
    for index11, index10 in pairs(function24) do
        if UNIQUE_TYPES[type(index10)] then
            if type(function24[index11]) ~= type(index10) then return false end
        elseif function25[index11] ~= index10 then
            return false
        end 
    end
    return true 
end

function1("WebSocket.connect", {}, function()
    local types = {
        Send = "function",
        Close = "function",
        OnMessage = { "table", "userdata" },
        OnClose = { "table", "userdata" }
    }
    local tbl = WebSocket.connect("ws://echo.websocket.events")
    assert(type(tbl) == "table" or type(tbl) == "userdata", "Did not return a table or userdata")
    for index2, index in pairs(types) do
        if type(index) == "table" then
            assert(table.find(index, type(tbl[index2])), "Did not return a " .. table.concat(index, ", ") .. " for " .. index2 .. " (a " .. type(tbl[index2]) .. ")")
        else
            assert(type(tbl[index2]) == index, "Did not return a " .. index .. " for " .. index2 .. " (a " .. type(tbl[index2]) .. ")")
        end 
    end
    tbl:Close() 
end)
